# 1. 디렉토리 생성
mkdir -p ~/miniconda3/envs/gaussian_autoencoder/etc/conda/activate.d
mkdir -p ~/miniconda3/envs/gaussian_autoencoder/etc/conda/deactivate.d

# 2. Activation 스크립트 생성
# Activation 스크립트
cat > ~/miniconda3/envs/gaussian_autoencoder/etc/conda/activate.d/env_vars.sh << 'EOF'
#!/bin/bash
export OLD_PATH=$PATH
export OLD_CPATH=$CPATH
export OLD_LD_LIBRARY_PATH=$LD_LIBRARY_PATH
export OLD_CUDA_HOME=$CUDA_HOME

export PATH=$CONDA_PREFIX/bin:$PATH
export CUDA_HOME=$CONDA_PREFIX
export CPATH=$CONDA_PREFIX/targets/x86_64-linux/include:$CPATH
export LD_LIBRARY_PATH=$CONDA_PREFIX/targets/x86_64-linux/lib:$LD_LIBRARY_PATH
EOF

# 3. Deactivation 스크립트 생성
cat > ~/miniconda3/envs/freetimegs/etc/conda/deactivate.d/env_vars.sh << 'EOF'
#!/bin/bash
if [ -n "$OLD_PATH" ]; then
    export PATH=$OLD_PATH
    unset OLD_PATH
fi
if [ -n "$OLD_CPATH" ]; then
    export CPATH=$OLD_CPATH
    unset OLD_CPATH
fi
if [ -n "$OLD_LD_LIBRARY_PATH" ]; then
    export LD_LIBRARY_PATH=$OLD_LD_LIBRARY_PATH
    unset OLD_LD_LIBRARY_PATH
fi
if [ -n "$OLD_CUDA_HOME" ]; then
    export CUDA_HOME=$OLD_CUDA_HOME
    unset OLD_CUDA_HOME
else
    unset CUDA_HOME
fi
EOF

# 4. 실행 권한 부여
chmod +x ~/miniconda3/envs/gaussian_autoencoder/etc/conda/activate.d/env_vars.sh
chmod +x ~/miniconda3/envs/gaussian_autoencoder/etc/conda/deactivate.d/env_vars.sh

# 5. 환경 재활성화
conda deactivate
conda activate gaussian_autoencoder
